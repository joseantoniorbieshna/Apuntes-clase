### TEMAS
```
create table temas(
Id INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
tema VARCHAR(255) UNIQUE);
```

```
DELIMITER @
CREATE OR REPLACE PROCEDURE GETTEMAS()
BEGIN
 SELECT * from temas;
END@
DELIMITER ;
```


```
DELIMITER @
CREATE OR REPLACE PROCEDURE PUTTEMA(IN DATA VARCHAR(255))
BEGIN
	Insert into temas(tema) values(DATA);
END@
DELIMITER ;
```
